
### 유사 배열

-   객체의 속성 명으로 인덱스(숫자)를 사용

**11\_arraylike.js**

```js
var human = {
    name: "김상형",
    age: 29
};
human[0] = 87;
human[1] = 79;
human[2] = 92;
human.length = 3;

for (var i = 0; i < human.length; i++) {
	console.log("human[" + i + "] =" + human[i]);
}

console.log("--------");

for (var i in human) {
	console.log("human[" + i + "] =" + human[i]);
}
```
